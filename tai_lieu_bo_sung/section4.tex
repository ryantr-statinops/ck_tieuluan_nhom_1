\section{The error estimates}\label{sec:error}

In this section, we will derive the unconditional error estimates of all the variables. Let
\begin{equation}\label{eq:error-def}
\begin{aligned}
  e_{u}^{j+1}       &= u_{h}^{j+1} - S_{h}u^{j+1}, \quad
  e_{p}^{j+1}        = p_{h}^{j+1} - S_{h}p^{j+1}, \\
  e_{\tilde{c}_{1}}^{j+1} &= \tilde{c}_{1h}^{j+1} - R_{h}\tilde{c}_{1}^{j+1}, \quad
  e_{\tilde{c}_{2}}^{j+1}  = \tilde{c}_{2h}^{j+1} - R_{h}\tilde{c}_{2}^{j+1}, \quad
  e_{\psi}^{j+1}     = \psi_{h}^{j+1} - R_{h}\psi^{j+1},
\end{aligned}
\end{equation}
and subtract \eqref{eq:scheme} from \eqref{eq:variational} with $t = t_{j+1}$, then we obtain the error equations
\begin{align}
  &\bigl(u_{t}^{j+1} - D_{\tau}u^{j+1},\, v_{h}\bigr)
   + \bigl(D_{\tau}(u^{j+1} - S_{h}u^{j+1}),\, v_{h}\bigr)
   - \bigl(D_{\tau}e_{u}^{j+1},\, v_{h}\bigr)
   - \bigl(\nabla e_{u}^{j+1},\, \nabla v_{h}\bigr) \nonumber\\
  &\quad + b_{1}(u^{j+1}-u^{j};\, u^{j+1},\, v_{h})
   + b_{1}(u^{j} - S_{h}u^{j};\, u^{j+1},\, v_{h})
   + b_{1}(S_{h}u^{j};\, u^{j+1} - S_{h}u^{j+1},\, v_{h}) \nonumber\\
  &\quad - b_{1}(e_{u}^{j};\, S_{h}u^{j+1},\, v_{h})
   - b_{1}(u_{h}^{j};\, e_{u}^{j+1},\, v_{h})
   + \bigl(e_{p}^{j+1},\, \operatorname{div} v_{h}\bigr) \nonumber\\
  &\quad + \bigl(\tilde{c}_{1}^{j+1}\nabla\psi^{j+1}
             - \tilde{c}_{1h}^{j}\nabla\psi_{h}^{j+1},\, v_{h}\bigr)
   - \bigl(\tilde{c}_{2}^{j+1}\nabla\psi^{j+1}
             - \tilde{c}_{2h}^{j}\nabla\psi_{h}^{j+1},\, v_{h}\bigr)
   = 0, \quad \forall\, v_{h}\in V_{h},
  \label{eq:err-u}\\[4pt]
  &\bigl(\operatorname{div} e_{u}^{j+1},\, q_{h}\bigr) = 0,
   \quad \forall\, q_{h}\in Q_{h},
  \label{eq:err-divu}\\[4pt]
  &\bigl(\tilde{c}_{1t}^{j+1} - D_{\tau}\tilde{c}_{1}^{j+1},\, \eta_{1h}\bigr)
   + \bigl(D_{\tau}\tilde{c}_{1}^{j+1} - D_{\tau}R_{h}\tilde{c}_{1}^{j+1},\, \eta_{1h}\bigr)
   - \bigl(D_{\tau}e_{\tilde{c}_{1}}^{j+1},\, \eta_{1h}\bigr)
   - \bigl(\nabla e_{\tilde{c}_{1}}^{j+1},\, \nabla\eta_{1h}\bigr) \nonumber\\
  &\quad - m_{0}\bigl(\nabla e_{\psi}^{j+1},\, \nabla\eta_{1h}\bigr)
   + \bigl((\tilde{c}_{1}^{j+1} - \tilde{c}_{1}^{j})\nabla\psi^{j+1},\, \nabla\eta_{1h}\bigr)
   + \bigl((\tilde{c}_{1}^{j} - R_{h}\tilde{c}_{1}^{j})\nabla\psi^{j+1},\, \nabla\eta_{1h}\bigr) \nonumber\\
  &\quad - \bigl(e_{\tilde{c}_{1}}^{j}\nabla\psi^{j+1},\, \nabla\eta_{1h}\bigr)
   + \bigl(\tilde{c}_{1h}^{j}\nabla(\psi^{j+1} - R_{h}\psi^{j+1}),\, \nabla\eta_{1h}\bigr)
   - \bigl(\tilde{c}_{1h}^{j}\nabla e_{\psi}^{j+1},\, \nabla\eta_{1h}\bigr) \nonumber\\
  &\quad + b_{2}(u^{j+1}-u^{j};\, \tilde{c}_{1}^{j+1},\, \eta_{1h})
   + b_{2}(u^{j} - S_{h}u^{j};\, \tilde{c}_{1}^{j+1},\, \eta_{1h})
   - b_{2}(e_{u}^{j};\, \tilde{c}_{1}^{j+1},\, \eta_{1h}) \nonumber\\
  &\quad + b_{2}(u_{h}^{j};\, \tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1},\, \eta_{1h})
   - b_{2}(u_{h}^{j};\, e_{\tilde{c}_{1}}^{j+1},\, \eta_{1h})
   = 0, \quad \forall\, \eta_{1h}\in X_{h},
  \label{eq:err-c1}\\[4pt]
  &\bigl(\tilde{c}_{2t}^{j+1} - D_{\tau}\tilde{c}_{2}^{j+1},\, \eta_{2h}\bigr)
   + \bigl(D_{\tau}\tilde{c}_{2}^{j+1} - D_{\tau}R_{h}\tilde{c}_{2}^{j+1},\, \eta_{2h}\bigr)
   - \bigl(D_{\tau}e_{\tilde{c}_{2}}^{j+1},\, \eta_{2h}\bigr)
   - \bigl(\nabla e_{\tilde{c}_{2}}^{j+1},\, \nabla\eta_{2h}\bigr) \nonumber\\
  &\quad + m_{0}\bigl(\nabla e_{\psi}^{j+1},\, \nabla\eta_{2h}\bigr)
   - \bigl((\tilde{c}_{2}^{j+1} - \tilde{c}_{2}^{j})\nabla\psi^{j+1},\, \nabla\eta_{2h}\bigr)
   - \bigl((\tilde{c}_{2}^{j} - R_{h}\tilde{c}_{2}^{j})\nabla\psi^{j+1},\, \nabla\eta_{2h}\bigr) \nonumber\\
  &\quad + \bigl(e_{\tilde{c}_{2}}^{j}\nabla\psi^{j+1},\, \nabla\eta_{2h}\bigr)
   - \bigl(\tilde{c}_{2h}^{j}\nabla(\psi^{j+1} - R_{h}\psi^{j+1}),\, \nabla\eta_{2h}\bigr)
   + \bigl(\tilde{c}_{2h}^{j}\nabla e_{\psi}^{j+1},\, \nabla\eta_{2h}\bigr) \nonumber\\
  &\quad + b_{2}(u^{j+1}-u^{j};\, \tilde{c}_{2}^{j+1},\, \eta_{2h})
   + b_{2}(u^{j} - S_{h}u^{j};\, \tilde{c}_{2}^{j+1},\, \eta_{2h})
   - b_{2}(e_{u}^{j};\, \tilde{c}_{2}^{j+1},\, \eta_{2h}) \nonumber\\
  &\quad + b_{2}(u_{h}^{j};\, \tilde{c}_{2}^{j+1} - R_{h}\tilde{c}_{2}^{j+1},\, \eta_{2h})
   - b_{2}(u_{h}^{j};\, e_{\tilde{c}_{2}}^{j+1},\, \eta_{2h})
   = 0, \quad \forall\, \eta_{2h}\in X_{h},
  \label{eq:err-c2}\\[4pt]
  &\bigl(\nabla e_{\psi}^{j+1},\, \nabla\varphi_{h}\bigr)
   = -\bigl(\tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1},\, \varphi_{h}\bigr)
   + \bigl(\tilde{c}_{2}^{j+1} - R_{h}\tilde{c}_{2}^{j+1},\, \varphi_{h}\bigr)
   + \bigl(e_{\tilde{c}_{1}}^{j+1} - e_{\tilde{c}_{2}}^{j+1},\, \varphi_{h}\bigr),
   \quad \forall\, \varphi_{h}\in X_{h}.
  \label{eq:err-psi}
\end{align}

% ------------------------------------------------------------
\subsection{The \texorpdfstring{$\ell^{\infty}(L^{2})$}{l-inf(L2)}-norm error estimates}
% ------------------------------------------------------------

Now, we derive the $\ell^{\infty}(L^{2})$-norm error estimates of all variables, which are shown in
following Lemmas~\ref{lem:epsi-l2}--\ref{lem:eu-l2} and Theorem~\ref{thm:l2}.

\begin{lemma}\label{lem:epsi-l2}
Assume that $(\tilde{c}_{1},\tilde{c}_{2},\psi,u,p)$ and
$(\tilde{c}_{1h}^{j+1},\tilde{c}_{2h}^{j+1},\psi_{h}^{j+1},u_{h}^{j+1},p_{h}^{j+1})$
are solutions of \eqref{eq:variational} and \eqref{eq:scheme}, respectively, and the regularity
condition \eqref{eq:regularity} holds. Then we have
\begin{equation}\label{eq:epsi-l2}
  \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0}
  + \bigl\|\Delta_{h} e_{\psi}^{j+1}\bigr\|_{0}
  \leq CC_{0}h^{2}
  + C\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
           + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}\Bigr).
\end{equation}
\end{lemma}

\begin{proof}
Taking $\varphi_{h} = e_{\psi}^{j+1}$ and $\varphi_{h} = \Delta_{h}e_{\psi}^{j+1}$
in \eqref{eq:err-psi}, respectively, we get
\begin{align}
  \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0}^{2}
  &\leq C\Bigl(\bigl\|\tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1}\bigr\|_{0}
              + \bigl\|\tilde{c}_{2}^{j+1} - R_{h}\tilde{c}_{2}^{j+1}\bigr\|_{0}\Bigr)
         \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0} \nonumber\\
  &\quad + C\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
                 + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}\Bigr)
            \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0},
  \label{eq:grad-epsi}\\[4pt]
  \bigl\|\Delta_{h}e_{\psi}^{j+1}\bigr\|_{0}^{2}
  &\leq \Bigl(\bigl\|\tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1}\bigr\|_{0}
             + \bigl\|\tilde{c}_{2}^{j+1} - R_{h}\tilde{c}_{2}^{j+1}\bigr\|_{0}\Bigr)
        \bigl\|\Delta_{h}e_{\psi}^{j+1}\bigr\|_{0} \nonumber\\
  &\quad + \Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
                + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}\Bigr)
           \bigl\|\Delta_{h}e_{\psi}^{j+1}\bigr\|_{0},
  \label{eq:Deltah-epsi}
\end{align}
which implies the desired result \eqref{eq:epsi-l2}.
\end{proof}

\begin{lemma}\label{lem:ec-l2}
Under the conditions of Lemma~\ref{lem:epsi-l2}, we have
\begin{align}
  &\bigl\|e_{\tilde{c}_{1}}^{J}\bigr\|_{0}^{2}
   + \bigl\|e_{\tilde{c}_{2}}^{J}\bigr\|_{0}^{2}
   + \tau\sum_{j=0}^{J-1}\Bigl(\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                               + \bigl\|\nabla e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}\Bigr)
  \nonumber\\
  &\qquad \leq CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})(\tau^{2}+h^{2})
   + C(C_{0}^{2}+C_{1}^{2})\,\tau\sum_{j=0}^{J-1}
     \Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
          + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}
          + \bigl\|e_{u}^{j+1}\bigr\|_{0}^{2}\Bigr).
  \label{eq:ec-l2}
\end{align}
\end{lemma}

\begin{proof}
Setting $\eta_{1h} = e_{\tilde{c}_{1}}^{j+1}$ in \eqref{eq:err-c1} leads to
\begin{align}
  &\frac{1}{2\tau}\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                       + \bigl\|e_{\tilde{c}_{1}}^{j+1} - e_{\tilde{c}_{1}}^{j}\bigr\|_{0}^{2}
                       - \bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}^{2}\Bigr)
   + \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
  \nonumber\\
  &= \bigl(\tilde{c}_{1t}^{j+1} - D_{\tau}\tilde{c}_{1}^{j+1},\, e_{\tilde{c}_{1}}^{j+1}\bigr)
   + \bigl(D_{\tau}\tilde{c}_{1}^{j+1} - D_{\tau}R_{h}\tilde{c}_{1}^{j+1},\, e_{\tilde{c}_{1}}^{j+1}\bigr)
   - m_{0}\bigl(\nabla e_{\psi}^{j+1},\, \nabla e_{\tilde{c}_{1}}^{j+1}\bigr)
  \nonumber\\
  &\quad + \bigl((\tilde{c}_{1}^{j+1}-\tilde{c}_{1}^{j})\nabla\psi^{j+1},\, \nabla e_{\tilde{c}_{1}}^{j+1}\bigr)
   + \bigl((\tilde{c}_{1}^{j} - R_{h}\tilde{c}_{1}^{j})\nabla\psi^{j+1},\, \nabla e_{\tilde{c}_{1}}^{j+1}\bigr)
   - \bigl(e_{\tilde{c}_{1}}^{j}\nabla\psi^{j+1},\, \nabla e_{\tilde{c}_{1}}^{j+1}\bigr)
  \nonumber\\
  &\quad + \bigl(\tilde{c}_{1h}^{j}\nabla(\psi^{j+1} - R_{h}\psi^{j+1}),\, \nabla e_{\tilde{c}_{1}}^{j+1}\bigr)
   - \bigl(\tilde{c}_{1h}^{j}\nabla e_{\psi}^{j+1},\, \nabla e_{\tilde{c}_{1}}^{j+1}\bigr)
  \nonumber\\
  &\quad + b_{2}(u^{j+1}-u^{j};\, \tilde{c}_{1}^{j+1},\, e_{\tilde{c}_{1}}^{j+1})
   + b_{2}(u^{j} - S_{h}u^{j};\, \tilde{c}_{1}^{j+1},\, e_{\tilde{c}_{1}}^{j+1})
   - b_{2}(e_{u}^{j};\, \tilde{c}_{1}^{j+1},\, e_{\tilde{c}_{1}}^{j+1})
  \nonumber\\
  &\quad + b_{2}(u_{h}^{j};\, \tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1},\, e_{\tilde{c}_{1}}^{j+1})
   \;:=\; \sum_{i=1}^{12} E_{i}.
  \label{eq:E-sum}
\end{align}
Now, we bound the above terms. From the Taylor formula, Lemma~\ref{lem:epsi-l2} and \eqref{eq:Ritz-error}, it is not hard to check that
\begin{align}
  \sum_{i=1}^{3}|E_{i}|
  &\leq C\tau\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{1tt}\|_{0}^{2}\,\mathrm{d}t
   + \frac{Ch^{4}}{\tau}\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{1t}\|_{2}^{2}\,\mathrm{d}t
   + CC_{0}^{2}h^{4} \nonumber\\
  &\quad + C\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                 + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}\Bigr)
   + \frac{1}{7}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2},
  \label{eq:E123}
\end{align}
and
\begin{align}
  \sum_{i=4}^{6}|E_{i}|
  &\leq \Bigl(\bigl\|\tilde{c}_{1}^{j+1}-\tilde{c}_{1}^{j}\bigr\|_{0}
             + \bigl\|\tilde{c}_{1}^{j} - R_{h}\tilde{c}_{1}^{j}\bigr\|_{0}
             + \bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}\Bigr)
    \bigl\|\nabla\psi^{j+1}\bigr\|_{0,\infty}
    \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\leq CC_{0}^{2}\!\left(\tau\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{1t}\|_{0}^{2}\,\mathrm{d}t
    + C_{0}^{2}h^{4}
    + \bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}^{2}\right)
   + \frac{1}{7}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:E456}
\end{align}
By use of the inductive hypothesis \eqref{eq:inductive}, Lemma~\ref{lem:Deltah-bound} and
Lemma~\ref{lem:epsi-l2}, we obtain
\begin{align}
  |E_{7}|
  &\leq \bigl\|\tilde{c}_{1h}^{j}\bigr\|_{0,3}
         \bigl\|\nabla(\psi^{j+1} - R_{h}\psi^{j+1})\bigr\|_{0,6}
         \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \leq CC_{0}^{2}C_{1}^{2}h^{2}
       + \frac{1}{7}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2},
  \label{eq:E7}\\[4pt]
  |E_{8}|
  &\leq \bigl\|\tilde{c}_{1h}^{j}\bigr\|_{0,3}
         \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0,6}
         \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \leq CC_{0}^{2}C_{1}^{2}h^{4}
       + CC_{1}^{2}\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                         + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}\Bigr)
       + \frac{1}{7}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:E8}
\end{align}
According to Lemma~\ref{lem:b2-bounds}, we infer
\begin{align}
  \sum_{i=9}^{11}|E_{i}|
  &\leq C\Bigl(\bigl\|u^{j+1}-u^{j}\bigr\|_{0}
              + \bigl\|u^{j} - S_{h}u^{j}\bigr\|_{0}
              + \bigl\|e_{u}^{j}\bigr\|_{0}\Bigr)
    \bigl\|\tilde{c}_{1}^{j+1}\bigr\|_{2}
    \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\leq CC_{0}^{2}\!\left(\tau\int_{t_{j}}^{t_{j+1}}\|u_{t}\|_{0}^{2}\,\mathrm{d}t
    + C_{0}^{2}h^{4}
    + \bigl\|e_{u}^{j}\bigr\|_{0}^{2}\right)
   + \frac{1}{7}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2},
  \label{eq:E9-11}
\end{align}
and
\begin{align}
  |E_{12}|
  &\leq \bigl|b_{2}(e_{u}^{j};\, \tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1},\,
                                   e_{\tilde{c}_{1}}^{j+1})\bigr|
       + \bigl|b_{2}(S_{h}u^{j};\, \tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1},\,
                                    e_{\tilde{c}_{1}}^{j+1})\bigr|
  \nonumber\\
  &\leq \bigl\|\nabla e_{u}^{j}\bigr\|_{0}
         \bigl\|\nabla(\tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1})\bigr\|_{0}
         \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
       + \|A_{h}S_{h}u^{j}\|_{0}
         \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
         \bigl\|\tilde{c}_{1}^{j+1} - R_{h}\tilde{c}_{1}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\leq CC_{0}^{2}\bigl\|e_{u}^{j}\bigr\|_{0}^{2}
       + CC_{0}^{4}h^{4}
       + \frac{1}{7}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:E12}
\end{align}
Inserting \eqref{eq:E123}--\eqref{eq:E12} into \eqref{eq:E-sum}, and summing up from $j = 0$
to $j = J-1$, we get
\begin{align}
  \bigl\|e_{\tilde{c}_{1}}^{J}\bigr\|_{0}^{2}
  + \tau\sum_{j=0}^{J-1}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
  &\leq CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})(\tau^{2}+h^{2})
  \nonumber\\
  &\quad + C(C_{0}^{2}+C_{1}^{2})\,\tau\sum_{j=0}^{J-1}
    \Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
         + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}
         + \bigl\|e_{u}^{j+1}\bigr\|_{0}^{2}\Bigr).
  \label{eq:ec1-l2-sum}
\end{align}
Subsequently, the similar estimates for $e_{\tilde{c}_{2}}^{j+1}$ can also be obtained,
which together with \eqref{eq:ec1-l2-sum} lead to \eqref{eq:ec-l2}.
\end{proof}

\begin{lemma}\label{lem:eu-l2}
Under the conditions of Lemma~\ref{lem:epsi-l2}, we have
\begin{align}
  \bigl\|e_{u}^{J}\bigr\|_{0}^{2}
  + \tau\sum_{j=0}^{J-1}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}
  &\leq CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})(\tau^{2}+h^{4})
  \nonumber\\
  &\quad + C(C_{0}^{2}+C_{1}^{2})\,\tau\sum_{j=0}^{J-1}
    \Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
         + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}
         + \bigl\|e_{u}^{j+1}\bigr\|_{0}^{2}\Bigr).
  \label{eq:eu-l2}
\end{align}
\end{lemma}

\begin{proof}
Choose $v_{h} = e_{u}^{j+1} \in V_{h}^{\mathrm{div}}$ in \eqref{eq:err-u}, and note that
\begin{equation}\label{eq:b1-cancel}
  b_{1}(u_{h}^{j};\, e_{u}^{j+1},\, e_{u}^{j+1}) = 0, \qquad
  \bigl(e_{p}^{j+1},\, \operatorname{div} e_{u}^{j+1}\bigr) = 0,
\end{equation}
then we arrive at
\begin{align}
  &\frac{1}{2\tau}\Bigl(\bigl\|e_{u}^{j+1}\bigr\|_{0}^{2}
                       + \bigl\|e_{u}^{j+1} - e_{u}^{j}\bigr\|_{0}^{2}
                       - \bigl\|e_{u}^{j}\bigr\|_{0}^{2}\Bigr)
   + \bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}
  \nonumber\\
  &= \bigl(u_{t}^{j+1} - D_{\tau}u^{j+1},\, e_{u}^{j+1}\bigr)
   + \bigl(D_{\tau}(u^{j+1} - S_{h}u^{j+1}),\, e_{u}^{j+1}\bigr)
   + b_{1}(u^{j+1}-u^{j};\, u^{j+1},\, e_{u}^{j+1})
  \nonumber\\
  &\quad + b_{1}(u^{j} - S_{h}u^{j};\, u^{j+1},\, e_{u}^{j+1})
   + b_{1}(S_{h}u^{j};\, u^{j+1} - S_{h}u^{j+1},\, e_{u}^{j+1})
   - b_{1}(e_{u}^{j};\, S_{h}u^{j+1},\, e_{u}^{j+1})
  \nonumber\\
  &\quad + \bigl(\tilde{c}_{1}^{j+1}\nabla\psi^{j+1}
              - \tilde{c}_{1h}^{j}\nabla\psi_{h}^{j+1},\, e_{u}^{j+1}\bigr)
   - \bigl(\tilde{c}_{2}^{j+1}\nabla\psi^{j+1}
              - \tilde{c}_{2h}^{j}\nabla\psi_{h}^{j+1},\, e_{u}^{j+1}\bigr)
   \;:=\; \sum_{i=1}^{8}G_{i}.
  \label{eq:G-sum}
\end{align}
By use of Taylor formula and Poincar\'{e} inequality, it is easy to check that
\begin{equation}\label{eq:G1}
  |G_{1}| \leq C\tau\int_{t_{j}}^{t_{j+1}}\|u_{tt}\|_{0}^{2}\,\mathrm{d}t
             + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}.
\end{equation}
Note that
\begin{align}
  \bigl\|D_{\tau}(u^{j+1} - S_{h}u^{j+1})\bigr\|_{0}^{2}
  &= \left\|\frac{(u^{j+1} - S_{h}u^{j+1}) - (u^{j} - S_{h}u^{j})}{\tau}\right\|_{0}^{2}
   = \left\|\frac{1}{\tau}\int_{t_{j}}^{t_{j+1}}(u - S_{h}u)_{t}\,\mathrm{d}t\right\|_{0}^{2}
  \nonumber\\
  &\leq \left\|\frac{1}{\tau}\left(\int_{t_{j}}^{t_{j+1}}1^{2}\,\mathrm{d}t\right)^{\!1/2}
         \left(\int_{t_{j}}^{t_{j+1}}\|(u - S_{h}u)_{t}\|_{0}^{2}\,\mathrm{d}t\right)^{\!1/2}
        \right\|_{0}^{2}
  \leq \frac{1}{\tau}\int_{t_{j}}^{t_{j+1}}\|(u - S_{h}u)_{t}\|_{0}^{2}\,\mathrm{d}t,
  \label{eq:Dtau-Shu}
\end{align}
then using \eqref{eq:Stokes-t-error}, we find
\begin{equation}\label{eq:G2}
  |G_{2}| \leq \frac{Ch^{4}}{\tau}\int_{t_{j}}^{t_{j+1}}
               \bigl(\|u_{t}\|_{2} + \|p_{t}\|_{1}\bigr)\,\mathrm{d}t
             + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}.
\end{equation}
Due to Lemma~\ref{lem:b1-bounds} and Taylor formula, we deduce
\begin{equation}\label{eq:G3G4}
  |G_{3}| + |G_{4}|
  \leq C\Bigl(\bigl\|u^{j+1}-u^{j}\bigr\|_{0} + \bigl\|u^{j} - S_{h}u^{j}\bigr\|_{0}\Bigr)
        \bigl\|u^{j+1}\bigr\|_{2}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}
  \leq CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}\|u_{t}\|_{0}^{2}\,\mathrm{d}t
     + CC_{0}^{4}h^{4}
     + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2},
\end{equation}
and
\begin{align}
  |G_{5}| &\leq C\|A_{h}S_{h}u^{j}\|_{0}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}
                  \bigl\|u^{j+1} - S_{h}u^{j+1}\bigr\|_{0}
           \leq CC_{0}^{4}h^{4} + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2},
  \label{eq:G5}\\[4pt]
  |G_{6}| &\leq C\bigl\|e_{u}^{j}\bigr\|_{0}\|A_{h}S_{h}u^{j+1}\|_{0}
                  \bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}
           \leq CC_{0}^{2}\bigl\|e_{u}^{j}\bigr\|_{0}^{2}
              + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:G6}
\end{align}
Next, $G_{7}$ can be split as
\begin{align}
  G_{7} &= \bigl((\tilde{c}_{1}^{j+1} - \tilde{c}_{1}^{j})\nabla\psi^{j+1},\, e_{u}^{j+1}\bigr)
          + \bigl(\tilde{c}_{1}^{j}\nabla(\psi^{j+1} - R_{h}\psi^{j+1}),\, e_{u}^{j+1}\bigr)
          + \bigl((\tilde{c}_{1}^{j} - R_{h}\tilde{c}_{1}^{j})\nabla R_{h}\psi^{j+1},\, e_{u}^{j+1}\bigr)
  \nonumber\\
  &\quad - \bigl(e_{\tilde{c}_{1}}^{j}\nabla R_{h}\psi^{j+1},\, e_{u}^{j+1}\bigr)
          - \bigl(\tilde{c}_{1h}^{j}\nabla e_{\psi}^{j+1},\, e_{u}^{j+1}\bigr)
         \;:=\; \sum_{i=1}^{5}G_{7i}.
  \label{eq:G7-split}
\end{align}
It is not hard to find that
\begin{align}
  |G_{71}| + |G_{73}| + |G_{74}| + |G_{75}|
  &\leq \bigl\|\tilde{c}_{1}^{j+1} - \tilde{c}_{1}^{j}\bigr\|_{0}
         \bigl\|\nabla\psi^{j+1}\bigr\|_{0,3}\bigl\|e_{u}^{j+1}\bigr\|_{0,6}
       + \bigl\|\tilde{c}_{1}^{j} - R_{h}\tilde{c}_{1}^{j}\bigr\|_{0}
         \bigl\|\nabla R_{h}\psi^{j+1}\bigr\|_{0,3}\bigl\|e_{u}^{j+1}\bigr\|_{0,6}
  \nonumber\\
  &\quad + \bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}
           \bigl\|\nabla R_{h}\psi^{j+1}\bigr\|_{0,3}\bigl\|e_{u}^{j+1}\bigr\|_{0,6}
         + \bigl\|\tilde{c}_{1h}^{j}\bigr\|_{0,3}
           \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0}\bigl\|e_{u}^{j+1}\bigr\|_{0,6}
  \nonumber\\
  &\leq CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{1t}\|_{0}^{2}\,\mathrm{d}t
    + CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})h^{4}
    + CC_{0}^{2}\bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}^{2}
  \nonumber\\
  &\quad + CC_{1}^{2}\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                           + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}\Bigr)
         + \frac{1}{16}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:G71-75}
\end{align}
Using Green formula, we obtain
\begin{align}
  |G_{72}|
  &= \bigl|\bigl(\psi^{j+1} - R_{h}\psi^{j+1},\,
                 \nabla\tilde{c}_{1}^{j}\cdot e_{u}^{j+1}\bigr)
          + \bigl(\psi^{j+1} - R_{h}\psi^{j+1},\,
                 \tilde{c}_{1}^{j}\operatorname{div}e_{u}^{j+1}\bigr)\bigr|
  \nonumber\\
  &\leq \bigl\|\psi^{j+1} - R_{h}\psi^{j+1}\bigr\|_{0}
    \Bigl(\bigl\|\nabla\tilde{c}_{1}^{j}\bigr\|_{0,3}\bigl\|e_{u}^{j+1}\bigr\|_{0,6}
         + \bigl\|\tilde{c}_{1}^{j}\bigr\|_{0,\infty}\bigl\|\operatorname{div}e_{u}^{j+1}\bigr\|_{0}\Bigr)
  \leq CC_{0}^{4}h^{4} + \frac{1}{16}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:G72}
\end{align}
Combining \eqref{eq:G71-75} and \eqref{eq:G72}, we have
\begin{align}
  |G_{7}|
  &\leq CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{1t}\|_{0}^{2}\,\mathrm{d}t
    + CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})h^{4}
    + CC_{0}^{2}\bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}^{2}
  \nonumber\\
  &\quad + CC_{1}^{2}\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                           + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}\Bigr)
         + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:G7}
\end{align}
In a similar way, $G_{8}$ can be estimated as
\begin{align}
  |G_{8}|
  &\leq CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{2t}\|_{0}^{2}\,\mathrm{d}t
    + CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})h^{4}
    + CC_{0}^{2}\bigl\|e_{\tilde{c}_{2}}^{j}\bigr\|_{0}^{2}
  \nonumber\\
  &\quad + CC_{1}^{2}\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                           + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}\Bigr)
         + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:G8}
\end{align}
Therefore, inserting estimates \eqref{eq:G1}, \eqref{eq:G2}--\eqref{eq:G6}, \eqref{eq:G7},
\eqref{eq:G8} into \eqref{eq:G-sum}, we get
\begin{align}
  &\frac{1}{2\tau}\Bigl(\bigl\|e_{u}^{j+1}\bigr\|_{0}^{2}
                       - \bigl\|e_{u}^{j}\bigr\|_{0}^{2}\Bigr)
   + \frac{1}{8}\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}
  \nonumber\\
  &\leq CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})h^{4}
   + \frac{Ch^{4}}{\tau}\int_{t_{j}}^{t_{j+1}}
     \bigl(\|u_{t}\|_{2}^{2} + \|p_{t}\|_{1}^{2}\bigr)\,\mathrm{d}t
   + CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}
     \bigl(\|u_{tt}\|_{0}^{2} + \|u_{t}\|_{0}^{2}
          + \|\tilde{c}_{1t}\|_{0}^{2} + \|\tilde{c}_{2t}\|_{0}^{2}\bigr)\,\mathrm{d}t
  \nonumber\\
  &\quad + CC_{0}^{2}\bigl\|e_{u}^{j}\bigr\|_{0}^{2}
   + C(C_{0}^{2}+C_{1}^{2})\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}^{2}
                                  + \bigl\|e_{\tilde{c}_{2}}^{j}\bigr\|_{0}^{2}
                                  + \bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                                  + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}\Bigr).
  \label{eq:eu-step}
\end{align}
Finally, summing \eqref{eq:eu-step} up from $j = 0$ to $J-1$, we arrive at \eqref{eq:eu-l2}.
\end{proof}

\begin{theorem}\label{thm:l2}
Under the conditions of Lemma~\ref{lem:epsi-l2}, if $\tau$ is small enough, we have
\begin{align}
  &\bigl\|e_{\tilde{c}_{1}}^{J}\bigr\|_{0}^{2}
   + \bigl\|e_{\tilde{c}_{2}}^{J}\bigr\|_{0}^{2}
   + \bigl\|e_{u}^{J}\bigr\|_{0}^{2}
   + \tau\sum_{j=0}^{J-1}\Bigl(\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                               + \bigl\|\nabla e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}
                               + \bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}\Bigr)
   \leq C_{2}^{2}\bigl(\tau^{2} + h^{2}\bigr),
  \label{eq:l2-result}\\[4pt]
  &\bigl\|\nabla e_{\psi}^{J}\bigr\|_{0}
   + \bigl\|\Delta_{h}e_{\psi}^{J}\bigr\|_{0}
   \leq C_{2}(\tau + h),
  \label{eq:l2-psi}
\end{align}
where $C_{2}^{2} = CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})\exp\!\bigl(CC_{0}^{2}+CC_{1}^{2}\bigr)$.
\end{theorem}

\begin{proof}
Using Lemma~\ref{lem:ec-l2} and Lemma~\ref{lem:eu-l2}, we get
\begin{align}
  &\bigl\|e_{\tilde{c}_{1}}^{J}\bigr\|_{0}^{2}
   + \bigl\|e_{\tilde{c}_{2}}^{J}\bigr\|_{0}^{2}
   + \bigl\|e_{u}^{J}\bigr\|_{0}^{2}
   + \tau\sum_{j=0}^{J-1}\Bigl(\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
                               + \bigl\|\nabla e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}
                               + \bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}\Bigr)
  \nonumber\\
  &\qquad \leq CC_{0}^{2}(C_{0}^{2}+C_{1}^{2})(\tau^{2}+h^{2})
   + C(C_{0}^{2}+C_{1}^{2})\,\tau\sum_{j=0}^{J-1}
     \Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}
          + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}^{2}
          + \bigl\|e_{u}^{j+1}\bigr\|_{0}^{2}\Bigr).
  \label{eq:combined-l2}
\end{align}
By the Gronwall lemma, taking $\tau < \tau_{2} := \dfrac{1}{C(C_{0}^{2}+C_{1}^{2})}$,
we deduce \eqref{eq:l2-result}. Then using Lemma~\ref{lem:epsi-l2}, we obtain \eqref{eq:l2-psi}.
\end{proof}

The error estimates in Lemmas~\ref{lem:epsi-l2}--\ref{lem:eu-l2} and Theorem~\ref{thm:l2}
were derived under the inductive hypothesis \eqref{eq:inductive}. Now, we check it.

When $j = 0$, we deduce
\begin{equation}\label{eq:j0-check}
  \bigl\|\tilde{c}_{1h}^{0}\bigr\|_{0,3}
  + \bigl\|\tilde{c}_{2h}^{0}\bigr\|_{0,3}
  = \|R_{h}\tilde{c}_{10}\|_{0,3} + \|R_{h}\tilde{c}_{20}\|_{0,3}
  < 1 + C_{0} =: C_{1}.
\end{equation}
Assume that
\begin{equation}\label{eq:inductive-m}
  \bigl\|\tilde{c}_{1h}^{m}\bigr\|_{0,3}
  + \bigl\|\tilde{c}_{2h}^{m}\bigr\|_{0,3} \leq C_{1}
\end{equation}
is true for $m \leq j$. Next, we derive \eqref{eq:inductive} for $m = j+1$. To this aim, we
consider two cases. If $\tau < h$, we have from the inverse inequality and Theorem~\ref{thm:l2} that
\begin{equation}\label{eq:case1}
  \bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0,3}
  + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0,3}
  \leq Ch^{-d/6}\Bigl(\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
                     + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}\Bigr)
  \leq CC_{2}h^{1/2}.
\end{equation}
If $h < \tau$, by Theorem~\ref{thm:l2}, we deduce
\begin{equation}\label{eq:case2}
  \bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0,3}
  + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0,3}
  \leq C\Bigl(\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
             + \bigl\|\nabla e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0}\Bigr)
  \leq CC_{2}\tau^{1/2}.
\end{equation}
In any case, if $\tau < \tau_{3} := \dfrac{1}{C^{2}C_{2}^{2}}$ and
$h < h_{1} := \dfrac{1}{C^{2}C_{2}^{2}}$, we conclude that
\begin{equation}\label{eq:err-small}
  \bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0,3}
  + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0,3} < 1.
\end{equation}
Thus, we have
\begin{equation}\label{eq:inductive-verified}
  \bigl\|\tilde{c}_{1h}^{j+1}\bigr\|_{0,3}
  + \bigl\|\tilde{c}_{2h}^{j+1}\bigr\|_{0,3}
  \leq \bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0,3}
     + \bigl\|e_{\tilde{c}_{2}}^{j+1}\bigr\|_{0,3}
     + \bigl\|R_{h}\tilde{c}_{1}^{j+1}\bigr\|_{0,3}
     + \bigl\|R_{h}\tilde{c}_{2}^{j+1}\bigr\|_{0,3}
  \leq 1 + C_{0} = C_{1}.
\end{equation}
The check of the inductive hypothesis \eqref{eq:inductive} is completed.

\begin{remark}\label{rem:inductive}
In the using of the mathematical inductive method, it is crucial to bound
$\|\tilde{c}_{1h}^{j+1}\|_{0,3}$ and $\|\tilde{c}_{2h}^{j+1}\|_{0,3}$ to an appointed
constant, otherwise the error bounds in Lemmas~\ref{lem:epsi-l2}--\ref{lem:eu-l2} and
Theorem~\ref{thm:l2} may become larger and larger.
\end{remark}

\begin{remark}\label{rem:suboptimal}
Theorem~\ref{thm:l2} shows that the error estimates of the $\ell^{\infty}(L^{2})$ norm are
suboptimal, which is caused by the fact that the estimate of the term $E_{7}$ in
Lemma~\ref{lem:ec-l2} cannot achieve optimal.
\end{remark}

Finally, we derive the unconditional stability of solutions.

\begin{lemma}\label{lem:stability}
Under the conditions of Lemma~\ref{lem:epsi-l2}, if $\tau$ and $h$ are small enough, we have
\begin{align}
  \bigl\|\nabla\psi_{h}^{j+1}\bigr\|_{0}
  + \bigl\|\Delta_{h}\psi_{h}^{j+1}\bigr\|_{0}
  &\leq C_{3},
  \label{eq:stab-psi}\\[4pt]
  \bigl\|\nabla\tilde{c}_{1h}^{j+1}\bigr\|_{0}
  + \bigl\|\nabla\tilde{c}_{2h}^{j+1}\bigr\|_{0}
  + \bigl\|\nabla u_{h}^{j+1}\bigr\|_{0}
  &\leq C_{3},
  \label{eq:stab-H1}\\[4pt]
  \tau\sum_{j=0}^{J-1}\Bigl(\bigl\|\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr\|_{0}^{2}
                            + \bigl\|\Delta_{h}\tilde{c}_{2h}^{j+1}\bigr\|_{0}^{2}
                            + \bigl\|A_{h}u_{h}^{j+1}\bigr\|_{0}^{2}\Bigr)
  &\leq C_{4}^{2},
  \label{eq:stab-H2}
\end{align}
where $C_{3} = CC_{2} + 2C_{0}$, $C_{4}^{2} = CC_{3}^{6} + CC_{0}^{2}$.
\end{lemma}

\begin{proof}
First of all, from the definition of the Ritz projection \eqref{eq:Ritz-def}, we have
\begin{equation}\label{eq:Ritz-prop}
  \bigl(\nabla R_{h}\psi^{j+1},\, \nabla\chi_{h}\bigr)
  = \bigl(\nabla\psi^{j+1},\, \nabla\chi_{h}\bigr), \quad \forall\,\chi_{h}\in X_{h}^{0}.
\end{equation}
Taking $\chi_{h} = R_{h}\psi^{j+1}$ and $\chi_{h} = \Delta_{h}R_{h}\psi^{j+1}$ in
\eqref{eq:Ritz-prop}, respectively, and using \eqref{eq:Deltah-def}, we infer
\begin{align}
  \bigl\|\nabla R_{h}\psi^{j+1}\bigr\|_{0}^{2}
  &= \bigl(\nabla\psi^{j+1},\, \nabla R_{h}\psi^{j+1}\bigr)
   \leq \bigl\|\nabla\psi^{j+1}\bigr\|_{0}\bigl\|\nabla R_{h}\psi^{j+1}\bigr\|_{0},
  \label{eq:Ritz-grad}\\[4pt]
  \bigl\|\Delta_{h}R_{h}\psi^{j+1}\bigr\|_{0}^{2}
  &= \bigl(\nabla R_{h}\psi^{j+1},\, \nabla\Delta_{h}R_{h}\psi^{j+1}\bigr)
   = \bigl(\nabla\psi^{j+1},\, \nabla\Delta_{h}R_{h}\psi^{j+1}\bigr)
  \nonumber\\
  &= -\bigl(\Delta\psi^{j+1},\, \Delta_{h}R_{h}\psi^{j+1}\bigr)
   \leq \bigl\|\Delta\psi^{j+1}\bigr\|_{0}\bigl\|\Delta_{h}R_{h}\psi^{j+1}\bigr\|_{0},
  \label{eq:Ritz-Deltah}
\end{align}
which implies
\begin{equation}\label{eq:Ritz-bound}
  \bigl\|\nabla R_{h}\psi^{j+1}\bigr\|_{0} \leq \bigl\|\nabla\psi^{j+1}\bigr\|_{0}, \qquad
  \bigl\|\Delta_{h}R_{h}\psi^{j+1}\bigr\|_{0} \leq \bigl\|\Delta\psi^{j+1}\bigr\|_{0}.
\end{equation}
Then using \eqref{eq:l2-psi} in Theorem~\ref{thm:l2}, we deduce
\begin{align}
  \bigl\|\nabla\psi_{h}^{j+1}\bigr\|_{0}
  + \bigl\|\Delta_{h}\psi_{h}^{j+1}\bigr\|_{0}
  &= \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0}
   + \bigl\|\Delta_{h}e_{\psi}^{j+1}\bigr\|_{0}
   + \bigl\|\nabla R_{h}\psi^{j+1}\bigr\|_{0}
   + \bigl\|\Delta_{h}R_{h}\psi^{j+1}\bigr\|_{0}
  \nonumber\\
  &\leq C_{2}(\tau+h) + 2\bigl\|\psi^{j+1}\bigr\|_{2}
   \leq CC_{2} + 2C_{0} =: C_{3}.
  \label{eq:stab-psi-proof}
\end{align}
Next, if $\tau < h$, from the inverse inequality and the $\ell^{\infty}(L^{2})$ norm estimate
in \eqref{eq:l2-result}, we infer
\begin{equation}\label{eq:grad-ec1-case1}
  \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \leq Ch^{-1}\bigl\|e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \leq CC_{2}h^{-1}(\tau+h) \leq CC_{2}.
\end{equation}
If $h < \tau$, by use of the $\ell^{2}(H^{1})$ norm estimate in \eqref{eq:l2-result}, we deduce
\begin{equation}\label{eq:grad-ec1-case2}
  \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \leq C_{2}\tau^{-1/2}(\tau+h) \leq CC_{2}\tau^{1/2}.
\end{equation}
In any case, we obtain
\begin{equation}\label{eq:stab-c1}
  \bigl\|\nabla\tilde{c}_{1h}^{j+1}\bigr\|_{0}
  \leq \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
     + \bigl\|\nabla R_{h}\tilde{c}_{1}^{j+1}\bigr\|_{0}
  \leq CC_{2} + C_{0} =: C_{3}.
\end{equation}
Similarly, we can derive the bounds of $\ell^{\infty}(H^{1})$ norm for
$\tilde{c}_{2h}^{j+1}$ and $u_{h}^{j+1}$, which implies \eqref{eq:stab-H1}.

Finally, taking $\eta_{1h} = 2\Delta_{h}\tilde{c}_{1h}^{j+1}\tau$ in the third equation
of \eqref{eq:scheme}, and using \eqref{eq:Deltah-def}, \eqref{eq:b1-b2-estimate3},
\eqref{eq:Lemma4-bound}, we have
\begin{align}
  &\bigl\|\nabla\tilde{c}_{1h}^{j+1}\bigr\|_{0}^{2}
   - \bigl\|\nabla\tilde{c}_{1h}^{j}\bigr\|_{0}^{2}
   + \bigl\|\nabla(\tilde{c}_{1h}^{j+1} - \tilde{c}_{1h}^{j})\bigr\|_{0}^{2}
   + 2\tau\bigl\|\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr\|_{0}^{2}
  \nonumber\\
  &\leq 2\tau\bigl(\tilde{c}_{1h}^{j}\nabla\psi_{h}^{j+1},\,
                   \nabla\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr)
       + 2\tau m_{0}\bigl(\nabla\psi_{h}^{j+1},\,
                           \nabla\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr)
       + 2\tau b_{2}\bigl(u_{h}^{j};\, \tilde{c}_{1h}^{j+1},\,
                           \Delta_{h}\tilde{c}_{1h}^{j+1}\bigr)
  \nonumber\\
  &\leq C\tau\Bigl(\bigl\|\nabla\tilde{c}_{1h}^{j}\bigr\|_{0}^{1/2}
                   \bigl\|\Delta_{h}\tilde{c}_{1h}^{j}\bigr\|_{0}^{1/2}
                  + m_{0}\Bigr)
    \bigl\|\Delta_{h}\psi_{h}^{j+1}\bigr\|_{0}
    \bigl\|\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\quad + C\tau\bigl\|\nabla u_{h}^{j}\bigr\|_{0}
    \bigl\|\nabla\tilde{c}_{1h}^{j+1}\bigr\|_{0}^{1/2}
    \bigl\|\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr\|_{0}^{3/2}
  \nonumber\\
  &\leq CC_{3}^{6}\tau + \tau\bigl\|\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:Deltah-c1h}
\end{align}
Summing \eqref{eq:Deltah-c1h} up from $j = 0$ to $J-1$, we get
\begin{equation}\label{eq:sum-Deltah-c1h}
  \tau\sum_{j=0}^{J-1}\bigl\|\Delta_{h}\tilde{c}_{1h}^{j+1}\bigr\|_{0}^{2}
  \leq CC_{3}^{6} + C\bigl\|\nabla\tilde{c}_{1h}^{0}\bigr\|_{0}^{2}
  \leq CC_{3}^{6} + CC_{0}^{2} =: C_{4}^{2}.
\end{equation}
In an analogous procedure, we can derive the rest part of \eqref{eq:stab-H2}. The proof is completed.
\end{proof}

\subsection{The \texorpdfstring{$\ell^{\infty}(H^{1})$}{l-inf(H1)}-norm error estimate}
% ------------------------------------------------------------

In this section, we derive the $\ell^{\infty}(H^{1})$-norm error estimates of all the variables,
which are shown in following Theorems~\ref{thm:H1-u}--\ref{thm:H1-all}.

\begin{theorem}\label{thm:H1-u}
Under the conditions of Lemma~\ref{lem:epsi-l2}, we have
\begin{equation}\label{eq:H1-u}
  \tau\sum_{j=0}^{J-1}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2}
  + \bigl\|\nabla e_{u}^{J}\bigr\|_{0}^{2}
  \leq C_{5}^{2}\bigl(\tau^{2} + h^{2}\bigr),
\end{equation}
where $C_{5}^{2} = C(C_{0}^{2}+C_{1}^{2})C_{2}^{2}\exp\!\bigl(CC_{0}^{2}+CC_{4}^{2}\bigr)$.
\end{theorem}

\begin{proof}
Putting $v_{h} = D_{\tau}e_{u}^{j+1}$ in \eqref{eq:err-u} results in
\begin{align}
  &\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2}
   + \frac{1}{2\tau}\Bigl(\bigl\|\nabla e_{u}^{j+1}\bigr\|_{0}^{2}
                         - \bigl\|\nabla e_{u}^{j}\bigr\|_{0}^{2}\Bigr)
  \nonumber\\
  &\leq \bigl(u_{t}^{j+1} - D_{\tau}u^{j+1},\, D_{\tau}e_{u}^{j+1}\bigr)
   + \bigl(D_{\tau}(u^{j+1} - S_{h}u^{j+1}),\, D_{\tau}e_{u}^{j+1}\bigr)
   + b_{1}(u^{j+1}-u^{j};\, u^{j+1},\, D_{\tau}e_{u}^{j+1})
  \nonumber\\
  &\quad + b_{1}(u^{j} - S_{h}u^{j};\, u^{j+1},\, D_{\tau}e_{u}^{j+1})
   + b_{1}(S_{h}u^{j};\, u^{j+1} - S_{h}u^{j+1},\, D_{\tau}e_{u}^{j+1})
   - b_{1}(e_{u}^{j};\, S_{h}u^{j+1},\, D_{\tau}e_{u}^{j+1})
  \nonumber\\
  &\quad - b_{1}(u_{h}^{j};\, e_{u}^{j+1},\, D_{\tau}e_{u}^{j+1})
   + \bigl(\tilde{c}_{1}^{j+1}\nabla\psi^{j+1}
             - \tilde{c}_{1h}^{j}\nabla\psi_{h}^{j+1},\, D_{\tau}e_{u}^{j+1}\bigr)
   - \bigl(\tilde{c}_{2}^{j+1}\nabla\psi^{j+1}
             - \tilde{c}_{2h}^{j}\nabla\psi_{h}^{j+1},\, D_{\tau}e_{u}^{j+1}\bigr)
   \;:=\; \sum_{i=1}^{9}L_{i}.
  \label{eq:L-sum}
\end{align}
Now, we estimate the above nine terms. It is not difficult to see that
\begin{equation}\label{eq:L1L2}
  |L_{1}| + |L_{2}|
  \leq C\tau\int_{t_{j}}^{t_{j+1}}\|u_{tt}\|_{0}^{2}\,\mathrm{d}t
     + \frac{Ch^{4}}{\tau}\int_{t_{j}}^{t_{j+1}}
       \bigl(\|u_{t}\|_{2}^{2} + \|p_{t}\|_{1}^{2}\bigr)\,\mathrm{d}t
     + \frac{1}{8}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2}.
\end{equation}
In view of Lemma~\ref{lem:b1-bounds}, we infer
\begin{align}
  |L_{3}| + |L_{4}|
  &\leq C\Bigl(\bigl\|\nabla(u^{j+1}-u^{j})\bigr\|_{0}
              + \bigl\|\nabla(u^{j} - S_{h}u^{j})\bigr\|_{0}\Bigr)
    \bigl\|u^{j+1}\bigr\|_{2}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\leq CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}\|\nabla u_{t}\|_{0}^{2}\,\mathrm{d}t
     + CC_{0}^{4}h^{2}
     + \frac{1}{8}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2},
  \label{eq:L3L4}
\end{align}
and
\begin{align}
  |L_{5}| &\leq \|A_{h}S_{h}u^{j}\|_{0}
                \bigl\|\nabla(u^{j+1} - S_{h}u^{j+1})\bigr\|_{0}
                \bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
           \leq CC_{0}^{4}h^{2}
              + \frac{1}{8}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2},
  \label{eq:L5}\\[4pt]
  |L_{6}| &\leq \bigl\|\nabla e_{u}^{j}\bigr\|_{0}
                \bigl\|A_{h}S_{h}u^{j+1}\bigr\|_{0}
                \bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
           \leq CC_{0}^{2}\bigl\|\nabla e_{u}^{j}\bigr\|_{0}^{2}
              + \frac{1}{8}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2},
  \label{eq:L6}\\[4pt]
  |L_{7}| &= \bigl|b_{1}(u_{h}^{j};\, e_{u}^{j+1} - e_{u}^{j},\, D_{\tau}e_{u}^{j+1})
              + b_{1}(u_{h}^{j};\, e_{u}^{j},\, D_{\tau}e_{u}^{j+1})\bigr|
  \nonumber\\
          &\leq 0
             + C\bigl\|A_{h}u_{h}^{j}\bigr\|_{0}
               \bigl\|\nabla e_{u}^{j}\bigr\|_{0}
               \bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
  \nonumber\\
          &\leq C\bigl\|A_{h}u_{h}^{j}\bigr\|_{0}^{2}
               \bigl\|\nabla e_{u}^{j}\bigr\|_{0}^{2}
             + \frac{1}{8}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:L7}
\end{align}
The term $L_{8}$ can be split as
\begin{align}
  L_{8} &= \bigl((\tilde{c}_{1}^{j+1} - \tilde{c}_{1}^{j})\nabla\psi^{j+1},\,
                 D_{\tau}e_{u}^{j+1}\bigr)
          + \bigl(\tilde{c}_{1}^{j}\nabla(\psi^{j+1} - R_{h}\psi^{j+1}),\,
                 D_{\tau}e_{u}^{j+1}\bigr)
  \nonumber\\
  &\quad + \bigl((\tilde{c}_{1}^{j} - R_{h}\tilde{c}_{1}^{j})\nabla R_{h}\psi^{j+1},\,
                 D_{\tau}e_{u}^{j+1}\bigr)
          - \bigl(e_{\tilde{c}_{1}}^{j}\nabla R_{h}\psi^{j+1},\,
                 D_{\tau}e_{u}^{j+1}\bigr)
          - \bigl(\tilde{c}_{1h}^{j}\nabla e_{\psi}^{j+1},\,
                 D_{\tau}e_{u}^{j+1}\bigr).
  \label{eq:L8-split}
\end{align}
Then we deduce
\begin{align}
  L_{8}
  &\leq \bigl\|\tilde{c}_{1}^{j+1} - \tilde{c}_{1}^{j}\bigr\|_{0}
         \bigl\|\nabla\psi^{j+1}\bigr\|_{0,\infty}
         \bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
       + \bigl\|\tilde{c}_{1}^{j}\bigr\|_{0,\infty}
         \bigl\|\nabla(\psi^{j+1} - R_{h}\psi^{j+1})\bigr\|_{0}
         \bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\quad + \Bigl(\bigl\|\tilde{c}_{1}^{j} - R_{h}\tilde{c}_{1}^{j}\bigr\|_{0}
               + \bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}\Bigr)
    \bigl\|\nabla R_{h}\psi^{j+1}\bigr\|_{0,\infty}
    \bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
  + \bigl\|\tilde{c}_{1h}^{j}\bigr\|_{0,3}
    \bigl\|\nabla e_{\psi}^{j+1}\bigr\|_{0,6}
    \bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\leq CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{1t}\|_{0}^{2}\,\mathrm{d}t
    + C(C_{0}^{2}+C_{1}^{2})C_{2}^{2}(\tau^{2}+h^{2})
    + \frac{1}{8}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:L8}
\end{align}
Similarly, we get
\begin{equation}\label{eq:L9}
  L_{9} \leq CC_{0}^{2}\tau\int_{t_{j}}^{t_{j+1}}\|\tilde{c}_{2t}\|_{0}^{2}\,\mathrm{d}t
           + C(C_{0}^{2}+C_{1}^{2})C_{2}^{2}(\tau^{2}+h^{2})
           + \frac{1}{8}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2}.
\end{equation}
Inserting estimates \eqref{eq:L1L2}--\eqref{eq:L9} into \eqref{eq:L-sum}, and then summing up
from $j = 0$ to $J-1$, we have
\begin{align}
  \tau\sum_{j=0}^{J-1}\bigl\|D_{\tau}e_{u}^{j+1}\bigr\|_{0}^{2}
  + \bigl\|\nabla e_{u}^{J}\bigr\|_{0}^{2}
  &\leq C(C_{0}^{2}+C_{1}^{2})C_{2}^{2}(\tau^{2}+h^{2})
   + C\tau\sum_{j=0}^{J-1}\Bigl(C_{0}^{2}
                                 + \bigl\|A_{h}u_{h}^{j}\bigr\|_{0}^{2}\Bigr)
    \bigl\|\nabla e_{u}^{j}\bigr\|_{0}^{2}.
  \label{eq:H1-u-pre}
\end{align}
Finally, by the Gronwall lemma and \eqref{eq:stab-H2}, we conclude \eqref{eq:H1-u}.
\end{proof}

\begin{theorem}\label{thm:H1-c}
Under the conditions of Lemma~\ref{lem:epsi-l2}, we have
\begin{equation}\label{eq:H1-c}
  \bigl\|\nabla e_{\tilde{c}_{1}}^{J}\bigr\|_{0}^{2}
  + \bigl\|\nabla e_{\tilde{c}_{2}}^{J}\bigr\|_{0}^{2}

Set $\eta_{1h} = D_{\tau}e_{\tilde{c}_{1}}^{j+1}$ in \eqref{eq:err-c1}, and we obtain

It is not hard to find that

\begin{equation}\label{eq:H1-c1-final}

A similar estimate can be obtained for $e_{\tilde{c}_{2}}^{j+1}$. The proof is completed.
\end{proof}

\subsection{Improved error estimate for the
  \texorpdfstring{$\ell^{\infty}(L^{2})$}{l-inf(L2)}-norm}
% ------------------------------------------------------------

In this section, let $\Omega$ be a rectangular domain with the boundary $\partial\Omega$
parallel to $x_{1}$-axis or $x_{2}$-axis. $\mathcal{T}_{h}$ is a family of axis-parallel
rectangular meshes of $\Omega$, and $K$ is a rectangular element. We use the bilinear element
to solve $c_{1}$, $c_{2}$, $\psi$, and $Q_{1}$-$Q_{0}$ element for $u$ and $p$
\cite{John2016}.

Then, we have the following improved estimate.

\begin{lemma}\label{lem:improved-l2}
For all $\psi\in H^{3}(\Omega)$, $\varphi_{h}\in X_{h}^{0}$, there holds
\begin{equation}\label{eq:improved-estimate}
  \bigl(\nabla(\psi - R_{h}\psi),\, \nabla\varphi_{h}\bigr)
  \leq Ch^{2}\|\psi\|_{3}\|\nabla\varphi_{h}\|_{0}.
\end{equation}
\end{lemma}

\begin{proof}
Let $I_{h}$ be the interpolation operator on $X_{h}^{0}$, then we obtain from
\cite{Lin2006} that
\begin{equation}\label{eq:Ih-estimate}
  \bigl(\nabla(\psi - I_{h}\psi),\, \nabla\varphi_{h}\bigr)
  \leq Ch^{2}\|\psi\|_{3}\|\nabla\varphi_{h}\|_{0}.
\end{equation}
By use of \eqref{eq:Ritz-def}, we have
\begin{equation}\label{eq:Ritz-Ih}
  \bigl(\nabla(R_{h}\psi - I_{h}\psi),\, \nabla\varphi_{h}\bigr)
  = \bigl(\nabla(\psi - I_{h}\psi),\, \nabla\varphi_{h}\bigr),
  \quad \forall\,\varphi_{h}\in X_{h}.
\end{equation}
Taking $\varphi_{h} = R_{h}\psi - I_{h}\psi$ in \eqref{eq:Ritz-Ih} and using
\eqref{eq:Ih-estimate}, we infer
\begin{equation}\label{eq:Ritz-Ih-bound}
  \bigl\|\nabla(R_{h}\psi - I_{h}\psi)\bigr\|_{0} \leq Ch^{2}\|\psi\|_{3}.
\end{equation}
As a result, from \eqref{eq:Ih-estimate} and \eqref{eq:Ritz-Ih-bound}, we deduce
\begin{align}
  \bigl(\nabla(\psi - R_{h}\psi),\, \nabla\varphi_{h}\bigr)
  &= \bigl(\nabla(\psi - I_{h}\psi),\, \nabla\varphi_{h}\bigr)
   + \bigl(\nabla(I_{h}\psi - R_{h}\psi),\, \nabla\varphi_{h}\bigr)
  \leq Ch^{2}\|\psi\|_{3}\|\nabla\varphi_{h}\|_{0}.
  \label{eq:improved-proof}
\end{align}
The proof is completed.
\end{proof}

Based on Lemma~\ref{lem:improved-l2}, we can improve the estimate of $E_{7}$ in
Lemma~\ref{lem:ec-l2}. In fact, $E_{7}$ can be split as
\begin{align}
  E_{7}
  &= \Bigl(\bigl(e_{\tilde{c}_{1}}^{j}
                + (R_{h}\tilde{c}_{1}^{j} - \tilde{c}_{1}^{j})
                + (\tilde{c}_{1}^{j} - P_{0}^{K}\tilde{c}_{1}^{j})\bigr)
           \nabla(\psi^{j+1} - R_{h}\psi^{j+1}),\,
           \nabla e_{\tilde{c}_{1}}^{j+1}\Bigr)
  \nonumber\\
  &\quad + \sum_{K} P_{0}^{K}\tilde{c}_{1}^{j}
    \Bigl(\nabla(\psi^{j+1} - \tilde{R}_{h}\psi^{j+1}),\,
          \nabla e_{\tilde{c}_{1}}^{j+1}\Bigr),
  \label{eq:E7-split}
\end{align}
where $P_{0}^{K}\tilde{c}_{1}^{j} = \dfrac{1}{|K|}\displaystyle\int_{K}\tilde{c}_{1}^{j}\,\mathrm{d}x$.

Then, using Lemma~\ref{lem:improved-l2}, we infer
\begin{align}
  |E_{7}|
  &\leq C\bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0,3}
         \bigl\|\nabla(\psi^{j+1} - R_{h}\psi^{j+1})\bigr\|_{0,6}
         \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
       + Ch^{2}\bigl\|\psi^{j+1}\bigr\|_{3}\bigl\|\tilde{c}_{1}^{j}\bigr\|_{2}
         \bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}
  \nonumber\\
  &\leq CC_{0}^{4}h^{4}
     + CC_{0}^{2}\bigl\|e_{\tilde{c}_{1}}^{j}\bigr\|_{0}^{2}
     + \frac{1}{7}\bigl\|\nabla e_{\tilde{c}_{1}}^{j+1}\bigr\|_{0}^{2}.
  \label{eq:E7-improved}
\end{align}
Consequently, we can improve the result of Theorem~\ref{thm:l2} as follows.

\begin{theorem}\label{thm:improved-l2}
Under the conditions of Lemma~\ref{lem:epsi-l2}, we have
\begin{equation}\label{eq:improved-l2-result}
  \bigl\|c_{1}^{j+1} - c_{1h}^{j+1}\bigr\|_{0}
  + \bigl\|c_{2}^{j+1} - c_{2h}^{j+1}\bigr\|_{0}
  + \bigl\|u^{j+1} - u_{h}^{j+1}\bigr\|_{0}
  \leq C_{2}\bigl(\tau + h^{2}\bigr).
\end{equation}
\end{theorem}

\begin{remark}\label{rem:higher-order}
Although only the first order spatial accuracy is reported in Theorem~\ref{thm:H1-all},
a higher order finite element can also be applied, and higher order error estimates can be
obtained under the condition of a higher regularity of solutions. In addition, on rectangular
meshes, there has been some existing numerical work about the super-convergence analysis of
the PNP equations and the Navier-Stokes equations \cite{Shi2019,Lin2006}. In a similar way,
we can obtain the super-convergence results for the NSPNP equations, and the convergence
order of the $\ell^{\infty}(H^{1})$-norm can be improved to $\mathcal{O}(\tau + h^{2})$.
\end{remark}
