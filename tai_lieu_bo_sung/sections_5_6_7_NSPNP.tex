\section{Two physical properties}
\label{sec:physical}
% ----------------------------------------------------------

In this section, we aim to derive some important physical properties for the proposed scheme.

\begin{theorem}[Global mass conservation]
\label{thm:mass}
The finite element solutions $\bigl(c_{1h}^{j+1},\, c_{2h}^{j+1}\bigr)$ satisfy global mass
conservation, that is,
\begin{equation}
  \int_{\Omega} c_{1h}^{j+1}\,\mathrm{d}x
  \;=\;
  \int_{\Omega} c_{2h}^{j+1}\,\mathrm{d}x
  \;=\;
  m_{0}\,|\Omega|.
  \label{eq:mass-cons}
\end{equation}
\end{theorem}

The property \eqref{eq:mass-cons} can be derived easily from \eqref{eq:step2}.

\begin{theorem}[Energy stability]
\label{thm:energy}
Define a discrete energy by
\begin{equation}
  E\!\left(u_{h}^{j},\,\psi_{h}^{j}\right)
  \;=\;
  \frac{1}{2}\left(
    \left\|u_{h}^{j}\right\|_{0}^{2}
    +
    \left\|\nabla\psi_{h}^{j}\right\|_{0}^{2}
  \right).
  \label{eq:energy-def}
\end{equation}
Then, if $\tau$ and $h$ are small enough, an energy stability
\begin{equation}
  E\!\left(u_{h}^{j+1},\,\psi_{h}^{j+1}\right)
  \;\leq\;
  E\!\left(u_{h}^{j},\,\psi_{h}^{j}\right)
  \label{eq:energy-stab}
\end{equation}
holds for $j = 0, 1, \ldots, J-1$.
\end{theorem}

\begin{proof}
\textit{Step 1.}
Choose $\varphi_{h} = \Delta_{h}\psi_{h}^{j+1}$ and
$\varphi_{h} = c_{1h}^{j+1} - c_{2h}^{j+1}$ in \eqref{eq:scheme} respectively, and we get
\begin{align}
  \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}^{2}
  &= \left(\nabla\psi_{h}^{j+1},\, \nabla\Delta_{h}\psi_{h}^{j+1}\right)
   = \left(c_{1h}^{j+1} - c_{2h}^{j+1},\, \Delta_{h}\psi_{h}^{j+1}\right)
   \nonumber\\
  &\leq
   \left\|c_{1h}^{j+1} - c_{2h}^{j+1}\right\|_{0}
   \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0},
   \label{eq:step1a}\\[4pt]
  \left\|c_{1h}^{j+1} - c_{2h}^{j+1}\right\|_{0}^{2}
  &= \left(\nabla\psi_{h}^{j+1},\, \nabla(c_{1h}^{j+1} - c_{2h}^{j+1})\right)
   = \left(c_{1h}^{j+1} - c_{2h}^{j+1},\, \Delta_{h}\psi_{h}^{j+1}\right)
   \nonumber\\
  &\leq
   \left\|c_{1h}^{j+1} - c_{2h}^{j+1}\right\|_{0}
   \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0},
   \label{eq:step1b}
\end{align}
which implies
\begin{align}
  \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}
  &= \left\|c_{1h}^{j+1} - c_{2h}^{j+1}\right\|_{0},
  \label{eq:Deltapsi-eq}\\
  \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}^{2}
  &= \left(\nabla(c_{1h}^{j+1} - c_{2h}^{j+1}),\,\nabla\psi_{h}^{j+1}\right).
  \label{eq:Deltapsi-sq}
\end{align}

\textit{Step 2.}
Putting $v_{h} = u_{h}^{j+1}$ in \eqref{eq:scheme} yields
\begin{equation}
  \frac{1}{2\tau}\!\left(
    \left\|u_{h}^{j+1}\right\|_{0}^{2}
    + \left\|u_{h}^{j+1} - u_{h}^{j}\right\|_{0}^{2}
    - \left\|u_{h}^{j}\right\|_{0}^{2}
  \right)
  + \left\|\nabla u_{h}^{j+1}\right\|_{0}^{2}
  = -\!\left((c_{1h}^{j} - c_{2h}^{j})\nabla\psi_{h}^{j+1},\, u_{h}^{j+1}\right).
  \label{eq:step2-u}
\end{equation}

\textit{Step 3.}
From the last equation of \eqref{eq:scheme}, it is easy to find
\begin{equation}
  \left(\nabla D_{\tau}\psi_{h}^{j+1},\, \nabla\varphi_{h}\right)
  = \left(D_{\tau}(c_{1h}^{j+1} - c_{2h}^{j+1}),\, \varphi_{h}\right),
  \quad \forall\,\varphi_{h} \in X_{h}.
  \label{eq:step3-psi}
\end{equation}
Take $\eta_{1h} = \eta_{2h} = \psi_{h}^{j+1}$ in \eqref{eq:scheme} and
$\varphi_{h} = \psi_{h}^{j+1}$ in \eqref{eq:step3-psi}, respectively; then we arrive at
\begin{align}
  &\frac{1}{2\tau}\!\left(
    \left\|\nabla\psi_{h}^{j+1}\right\|_{0}^{2}
    + \left\|\nabla(\psi_{h}^{j+1} - \psi_{h}^{j})\right\|_{0}^{2}
    - \left\|\nabla\psi_{h}^{j}\right\|_{0}^{2}
  \right)
  + \left(\nabla(c_{1h}^{j+1} - c_{2h}^{j+1}),\, \nabla\psi_{h}^{j+1}\right)
  \nonumber\\
  &\quad
  + \left((c_{1h}^{j} + c_{2h}^{j})\nabla\psi_{h}^{j+1},\, \nabla\psi_{h}^{j+1}\right)
  = -b_{2}\!\left(u_{h}^{j};\, c_{1h}^{j+1} - c_{2h}^{j+1},\, \psi_{h}^{j+1}\right).
  \label{eq:step3-sum}
\end{align}

\textit{Step 4.}
Adding \eqref{eq:step2-u} and \eqref{eq:step3-sum} and using \eqref{eq:Deltapsi-sq}, we deduce
\begin{align}
  &\frac{1}{2\tau}\!\Bigl(
    \left\|u_{h}^{j+1}\right\|_{0}^{2}
    + \left\|u_{h}^{j+1} - u_{h}^{j}\right\|_{0}^{2}
    - \left\|u_{h}^{j}\right\|_{0}^{2}
    + \left\|\nabla\psi_{h}^{j+1}\right\|_{0}^{2}
    + \left\|\nabla(\psi_{h}^{j+1} - \psi_{h}^{j})\right\|_{0}^{2}
    - \left\|\nabla\psi_{h}^{j}\right\|_{0}^{2}
  \Bigr)
  \nonumber\\
  &\quad
  + \left\|\nabla u_{h}^{j+1}\right\|_{0}^{2}
  + \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}^{2}
  + \left\|\sqrt{c_{1}^{j} + c_{2}^{j}}\,\nabla\psi_{h}^{j+1}\right\|_{0}^{2}
  \nonumber\\
  &= -\!\left((c_{1h}^{j} - c_{1}^{j} + c_{2h}^{j} - c_{2}^{j})\nabla\psi_{h}^{j+1},\,
              \nabla\psi_{h}^{j+1}\right)
  \nonumber\\
  &\quad
  + \left((c_{1h}^{j+1} - c_{1h}^{j} - (c_{2h}^{j+1} - c_{2h}^{j}))\nabla\psi_{h}^{j+1},\,
         u_{h}^{j+1}\right)
  \nonumber\\
  &\quad
  - \left((c_{1h}^{j+1} - c_{2h}^{j+1})\nabla\psi_{h}^{j+1},\, u_{h}^{j+1}\right)
  \nonumber\\
  &\quad
  + b_{2}\!\left(u_{h}^{j+1} - u_{h}^{j};\, c_{1h}^{j+1} - c_{2h}^{j+1},\, \psi_{h}^{j+1}\right)
  - b_{2}\!\left(u_{h}^{j+1};\, c_{1h}^{j+1} - c_{2h}^{j+1},\, \psi_{h}^{j+1}\right)
  \nonumber\\
  &=: \sum_{i=1}^{5} R_{i}.
  \label{eq:step4-Ri}
\end{align}

Next, we estimate the above five terms. By use of Theorem~\ref{thm:L2error}, it is easy to check
\begin{equation}
  \left\|c_{1h}^{j+1} - c_{1h}^{j}\right\|_{0}
  \leq
  \left\|c_{1h}^{j+1} - c_{1}^{j+1}\right\|_{0}
  + \left\|c_{1h}^{j} - c_{1}^{j}\right\|_{0}
  + \left\|c_{1}^{j+1} - c_{1}^{j}\right\|_{0}
  \leq C_{2}(\tau + h).
  \label{eq:c1-diff}
\end{equation}
Similarly, we obtain
\begin{equation}
  \left\|c_{2h}^{j+1} - c_{2h}^{j}\right\|_{0}
  \leq C_{2}(\tau + h),
  \qquad
  \left\|\nabla\!\left(u_{h}^{j+1} - u_{h}^{j}\right)\right\|_{0}
  \leq C_{5}(\tau + h).
  \label{eq:u-diff}
\end{equation}
Then, based on \eqref{eq:Deltapsi-eq}, \eqref{eq:c1-diff}, \eqref{eq:u-diff} and
Theorem~\ref{thm:L2error}, we infer
\begin{align}
  |R_{1}| + |R_{2}| + |R_{4}|
  &\leq
  \left(\left\|c_{1h}^{j} - c_{1}^{j}\right\|_{0}
        + \left\|c_{2h}^{j} - c_{2}^{j}\right\|_{0}\right)
  \left\|\nabla\psi_{h}^{j+1}\right\|_{0,3}
  \left\|\nabla\psi_{h}^{j+1}\right\|_{0,6}
  \nonumber\\
  &\quad
  + \left(\left\|c_{1h}^{j+1} - c_{1h}^{j}\right\|_{0}
          + \left\|c_{2h}^{j+1} - c_{2h}^{j}\right\|_{0}\right)
  \left\|\nabla\psi_{h}^{j+1}\right\|_{0,3}
  \left\|u_{h}^{j+1}\right\|_{0,6}
  \nonumber\\
  &\quad
  + \left\|\nabla\!\left(u_{h}^{j+1} - u_{h}^{j}\right)\right\|_{0}
  \left\|c_{1h}^{j+1} - c_{2h}^{j+1}\right\|_{0}
  \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}
  \nonumber\\
  &\leq
  C_{5}(\tau + h)\left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}^{2}
  + \frac{1}{2}\left\|\nabla u_{h}^{j+1}\right\|_{0}^{2}.
  \label{eq:R1R2R4}
\end{align}
Using Green's formula, Lemma~\ref{lem:1} (equation \eqref{eq:Lem1}), Theorem~\ref{thm:L2error}
and \eqref{eq:Deltapsi-eq}, we get
\begin{align}
  |R_{3} + R_{5}|
  &= \frac{1}{2}
     \left|\left(\mathrm{div}\,u_{h}^{j+1},\,
                 (c_{1h}^{j+1} - c_{2h}^{j+1})\psi_{h}^{j+1}\right)\right|
  \nonumber\\
  &= \frac{1}{2}
     \left|\left(\mathrm{div}(u_{h}^{j+1} - u^{j+1}),\,
                 (c_{1h}^{j+1} - c_{2h}^{j+1})\psi_{h}^{j+1}\right)\right|
  \nonumber\\
  &\leq
  C_{5}(\tau + h)
  \left\|c_{1h}^{j+1} - c_{2h}^{j+1}\right\|_{0}
  \left\|\psi_{h}^{j+1}\right\|_{0,\infty}
  \leq
  C_{5}(\tau + h)\left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}^{2}.
  \label{eq:R3R5}
\end{align}
Inserting \eqref{eq:R1R2R4} and \eqref{eq:R3R5} into \eqref{eq:step4-Ri} yields
\begin{align}
  &\frac{1}{2}\!\left(
    \left\|u_{h}^{j+1}\right\|_{0}^{2}
    + \left\|\nabla\psi_{h}^{j+1}\right\|_{0}^{2}
  \right)
  + \frac{1}{2}\!\left(
    \left\|u_{h}^{j+1} - u_{h}^{j}\right\|_{0}^{2}
    + \left\|\nabla(\psi_{h}^{j+1} - \psi_{h}^{j})\right\|_{0}^{2}
  \right)
  \nonumber\\
  &\quad
  + \tau\!\left(1 - C_{5}(\tau + h)\right)
    \left\|\Delta_{h}\psi_{h}^{j+1}\right\|_{0}^{2}
  + \frac{1}{2}\,\tau\left\|\nabla u_{h}^{j+1}\right\|_{0}^{2}
  + \tau\left\|\sqrt{c_{1}^{j} + c_{2}^{j}}\,\nabla\psi_{h}^{j+1}\right\|_{0}^{2}
  \nonumber\\
  &\leq
  \frac{1}{2}\!\left(
    \left\|u_{h}^{j}\right\|_{0}^{2}
    + \left\|\nabla\psi_{h}^{j}\right\|_{0}^{2}
  \right).
  \label{eq:energy-ineq}
\end{align}
If $\tau$ and $h$ are small enough, \eqref{eq:energy-stab} holds.
\end{proof}

\section{Numerical implementation}
\label{sec:numerics}
% ----------------------------------------------------------

In this section, we present three numerical examples to verify the theoretical results. The
first two examples have smooth exact solutions, which are used to confirm the convergence
orders. The third example is to show the stability of the scheme.

% ---- Example 1 --------------------------------------------
\begin{example}
\label{ex:1}
The first example refers to \cite{Gao2017} with some modifications. Take
$\Omega = [0,1]\times[0,1]$, and consider the example with following exact solutions
\begin{align}
  u_{1} &= -e^{-t}\sin^{2}(\pi x_{1})\sin(\pi x_{2})\cos(\pi x_{2}),
  \label{eq:ex1-u1}\\
  u_{2} &= \phantom{-}e^{-t}\sin(\pi x_{1})\cos(\pi x_{1})\sin^{2}(\pi x_{2}),
  \label{eq:ex1-u2}\\
  p     &= \sin(\pi x_{1})\cos(\pi x_{2}),
  \label{eq:ex1-p}\\
  c_{1} &= \frac{1}{2}\,e^{-t}\!\left(\cos(\pi x_{1})\cos(\pi x_{2}) + 1\right),
  \label{eq:ex1-c1}\\
  c_{2} &= \frac{1}{2}\,e^{-t}\!\left(\cos(2\pi x_{1})\cos(2\pi x_{2}) + 1\right),
  \label{eq:ex1-c2}\\
  \psi  &= \frac{1}{2\pi^{2}}\,e^{-t}\!\left(
             \cos(\pi x_{1})\cos(\pi x_{2})
             - \frac{1}{4}\cos(2\pi x_{1})\cos(2\pi x_{2})
           \right).
  \label{eq:ex1-psi}
\end{align}
We use two pairs of elements in the computation. The first pair (Pair~I) employs the linear
element to approximate $c_{1}$, $c_{2}$, $\psi$, and the MINI element for $u$ and $p$. The
second pair (Pair~II) uses the rectangular bilinear/$Q_{1}$-$Q_{0}$ element described in
Section~\ref{sec:improved}. The errors and convergence orders at $T = 1$ are shown in
Tables~\ref{tab:PairI-c1}--\ref{tab:PairII-u}.
\end{example}

From Tables~\ref{tab:PairI-c1}--\ref{tab:PairII-u}, we can see that the $l^{\infty}(L^{2})$
and $l^{\infty}(H^{1})$-norm errors of all variables are convergent at rates of
$\mathcal{O}(h^{2})$ and $\mathcal{O}(h)$ respectively for the two finite element pairs. It
is clear that all variables achieve optimal convergence order, which accords with our
theoretical analysis completely.

In addition, it seems that the errors of the approximate solutions obtained using Pair~II are
less than those of Pair~I. Moreover, Pair~II has fewer degrees of freedom than Pair~I, which
means less computation time. However, Pair~II is the rectangular element and Pair~I is the
triangular element, so the latter has better adaptability for the computational domain.

\section{Conclusion and future work}
\label{sec:conclusion}
% ----------------------------------------------------------

In this work, we present a linearized backward Euler finite element scheme for the NSPNP
equations, and prove the existence and uniqueness of this scheme. We then obtain the
unconditional error estimates of all variables in $l^{\infty}(L^{2})$ and
$l^{\infty}(H^{1})$ norms. Moreover, we give two important physical properties: the global
mass conservation and energy stability.

However, the concentration positivity cannot be theoretically justified, which is also
important for the numerical scheme of the NSPNP equations. In fact, for the pure PNP
equations, there has been some existing numerical methods which keep these physical
properties. In the future, we will try to apply these methods to the NSPNP equations. In
addition, our numerical scheme is only first-order accurate in time, and in a forthcoming
research we plan to propose some second-order accurate numerical schemes for the NSPNP
equations, such as the Crank--Nicolson scheme and the BDF2 scheme. Finally, it is
interesting and challenging to construct robust schemes to solve some practical problems,
such as charging dynamics in a nanopore, electroosmotic flow in a nanochannel, and
electroconvective flow on an ion-selective surface \cite{Wang2017}, and so on.
